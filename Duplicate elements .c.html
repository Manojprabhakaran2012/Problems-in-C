#include #include void findDuplicates(int arr[], int n) { for (int i = 0; i < n; i++) { for (int j = i + 1; j < n; j++) { if (arr[i] == arr[j]) { printf("Duplicate: %d\n", arr[i]); } } } } int main() { int arr[] = {1, 2, 3, 4, 2, 5, 3}; int n = sizeof(arr) / sizeof(arr[0]); findDuplicates(arr, n); return 0; }
